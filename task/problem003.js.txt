class Tree {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

function minimumPath(tree) {
    if (tree === null) {
        return Number.MAX_SAFE_INTEGER;
    }

 
    if (tree.left === null && tree.right === null) {
        return tree.val;
    }

    const left = minimumPath(tree.left);
    const right = minimumPath(tree.right);

    return tree.val + Math.min(left, right);
}


const t = new Tree(10);
t.left = new Tree(5);
t.right = new Tree(5);

t.left.right = new Tree(2);
t.right.right = new Tree(1);
t.right.right.left = new Tree(-1);

console.log(minimumPath(t));
