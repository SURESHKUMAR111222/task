class Tree {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

function invertTree(tree) {
    if (tree === null) {
        return null;
    }

    // Swap left and right children
    const temp = tree.left;
    tree.left = tree.right;
    tree.right = temp;

    invertTree(tree.left);
    invertTree(tree.right);

    return tree;
}

function printOrder(tree) {
    if (tree === null) return;

    process.stdout.write(tree.val + " "); // Pre-order print
    printOrder(tree.left);
    printOrder(tree.right);
}

// Main equivalent
function main() {
    const a = new Tree('a');
    const b = new Tree('b');
    const c = new Tree('c');
    const d = new Tree('d');
    const e = new Tree('e');
    const f = new Tree('f');

    a.left = b;
    a.right = c;
    b.left = d;
    b.right = e;
    c.left = f;

    invertTree(a);
    printOrder(a);
}

main();
