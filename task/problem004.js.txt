function kaprekarNumber(num) {
    return num >= 1000 && num <= 9999;
}
function countStep(num) {
    let count = 0;

    while (num !== 6174) {
        const str = num.toString().padStart(4, "0");
        let arr = str.split("").sort();
         const smallString = arr.join("");
        arr.reverse();
        const largeString = arr.join("");
        const smallNum = parseInt(smallString, 10);
        const largeNum = parseInt(largeString, 10);
        num = largeNum - smallNum;
        count++;
    }
    return count;
}

function main() {
    const readline = require("readline").createInterface({
        input: process.stdin,
        output: process.stdout
    });

    readline.question("Enter the number: ", input => {
        const number = parseInt(input, 10);

        if (!kaprekarNumber(number)) {
            console.log("Enter a valid number");
        } else {
            console.log("Total steps " + countStep(number));
        }

        readline.close();
    });
}

main();
